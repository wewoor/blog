Strut2 与 Hibernate 的一个 web 分页功能
2012-09-06 11:32 folder [计算机科学]  

代码没有进行过多的封装，可能看起来有点 action 代码部分，hibernate 在这里只起到了一个查询记录集的作用。

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

import org.apache.struts2.interceptor.SessionAware;
import org.hibernate.Criteria;
import org.hibernate.Session;

import com.HibernateSessionFactory;
import com.hibertest1.dto.MyMessage;

public class FenYeAction implements SessionAware{

private int start = 0;

public int getStart() {
return start;
}

public void setStart(int start) {
this.start = start;
}

public String getMessage(){

List<MyMessage> msgs = new ArrayList<MyMessage>();

Session se = HibernateSessionFactory.getSession();

Criteria crit = se.createCriteria(MyMessage.class); //创建Criteria 对象

msgs = crit.list();

int size = msgs.size(); //获得查询出来的总量

   if(size>5)

   {
           session.put("pageCount",size%5);

          if(size-start<5)    //如果起点位置到记录结束位置在5条结果之内
          {
                    session.put("subMsg",msgs.subList(start, size));            //那么我们就取值到结束点的位置
           }else
           {
                    session.put("subMsg",msgs.subList(start, start+5));            //否则一次获取5条记录来
           }

 }
  else
  {
                session.put("subMsg", msgs);             //如果小于等于5条记录就全部放入session
   }

return "succ";
}

private Map<String,Object> session;
public void setSession(Map<String, Object> arg0) {
// TODO Auto-generated method stub
this.session = arg0;
}
